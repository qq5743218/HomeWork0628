CREATE DATABASE test0329 DEFAULT CHARACTER SET 'utf8' DEFAULT COLLATE 'utf8_general_ci';
CREATE USER 'mememe'@'localhost' IDENTIFIED BY '123456';
GRANT ALL PRIVILEGES ON test0329.* TO 'mememe'@'localhost';
FLUSH PRIVILEGES;

USE test0329;

CREATE TABLE kanban (
	k_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	k_name VARCHAR(10),
	k_content VARCHAR(200)
) CHARACTER SET 'utf8' COLLATE 'utf8_general_ci';

CREATE TABLE forum(
	k_id INT UNSIGNED NOT NULL,
	f_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	f_content VARCHAR(200),
	FOREIGN KEY (k_id) REFERENCES kanban(k_id)
) CHARACTER SET 'utf8' COLLATE 'utf8_general_ci';
